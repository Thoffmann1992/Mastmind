    <script>
        const form = document.getElementById('remasterForm');
        const status = document.getElementById('status');
        const downloadBtn = document.getElementById('download-btn');

        let backendUrl = 'https://your-remasterify-backend.onrender.com/remaster';  // ← BURANI ÖZ BACKEND URL-İN İLƏ ƏVƏZ ET!

        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            status.textContent = 'Fayl yüklənir və emal olunur... (30 saniyə – 2 dəqiqə)';
            status.style.color = '#3498db';
            downloadBtn.style.display = 'none';

            const formData = new FormData();
            formData.append('file', document.getElementById('file').files[0]);
            formData.append('model', document.getElementById('model').value);
            formData.append('lufs', document.getElementById('lufs').value);
            formData.append('width', document.getElementById('width').value);
            formData.append('noise', document.getElementById('noise').value);

            try {
                const response = await fetch(backendUrl, {
                    method: 'POST',
                    body: formData
                });

                if (!response.ok) {
                    throw new Error(`Server xətası: ${response.status}`);
                }

                const blob = await response.blob();
                const url = URL.createObjectURL(blob);
                
                downloadBtn.href = url;  // əgər <a> tag istifadə etsəydik, amma button olaraq saxlayırıq
                downloadBtn.onclick = () => {
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = 'remastered_audio.wav';
                    document.body.appendChild(a);
                    a.click();
                    document.body.removeChild(a);
                };
                downloadBtn.style.display = 'block';
                downloadBtn.textContent = 'Remaster edilmiş faylı endir 🎧';

                status.textContent = 'Hazır! Aşağıdakı düymədən endirin.';
                status.style.color = '#27ae60';
            } catch (err) {
                status.textContent = 'Xəta: ' + err.message;
                status.style.color = '#e74c3c';
            }
        });
    </script>
